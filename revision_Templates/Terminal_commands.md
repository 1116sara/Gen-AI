# Navigation
cd foldername          # go into folder
cd ..                  # go up one level
cd D:\full\path        # go to specific path

# See where you are
cd                     # prints current directory

# See what's in current folder
dir                    # list files and folders
dir /a                 # include hidden files

# Find a file
dir /s filename.ext    # search recursively

# Run Python
python filename.py     # run a script

# Package management
uv pip install package # install package
uv pip list            # see installed packages

# Environment
.venv\Scripts\activate # activate virtual environment
